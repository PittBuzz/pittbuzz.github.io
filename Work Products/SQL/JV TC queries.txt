/*POM HPI Payouts (Factor) by SM (Driver) and FY*/
SELECT
	MDEP
	,APC_Initial
	,APC_Anniversary
	,IncentiveType
	,BonusGroup
	,BonusType_ID
	,RateDescription
	,ContractFY
	,sum(distinct case when TotalBonusDfasPaid is not null and ContractFY = '2021' then TotalBonusDfasPaid else 0 end) as FY21Factor
	,count(distinct case when TotalBonusDfasPaid is not null and ContractFY = '2021' then ssnid else 0 end) as FY21Driver
	,sum(distinct case when TotalBonusDfasPaid is not null and ContractFY = '2022' then TotalBonusDfasPaid else 0 end) as FY22Factor
	,count(distinct case when TotalBonusDfasPaid is not null and ContractFY = '2022' then ssnid else 0 end) as FY22Driver
	,sum(distinct case when TotalBonusDfasPaid is not null and ContractFY = '2023' then TotalBonusDfasPaid else 0 end) as FY23Factor
	,count(distinct case when TotalBonusDfasPaid is not null and ContractFY = '2023' then ssnid else 0 end) as FY23Driver

FROM POMMaster
WHERE ContractFY  > '2016'
AND IncentiveType in ('slrp','clrp')
Group By MDEP
	,APC_Initial
	,APC_Anniversary
	,IncentiveType
	,BonusGroup
	,BonusType_ID
	,RateDescription
	,ContractFY
Order by 8,5,6
